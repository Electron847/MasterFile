001    package agent.one;
002    import agent.Agent;
003    import agent.TimeServer;
004    import agent.TimeServerLinked;
005    
006    public class Main {
007      public static void main (String[] args) {
008        TimeServer time = new TimeServerLinked();
009        Agent a = new Tiger(time);
010        time.enqueue(0,a);
011        time.run(100);
012      }
013    }
014    
015    class Tiger implements Agent {
016      private TimeServer _time;
017      public Tiger(TimeServer time) { _time = time; }
018      public void run() {
019        System.out.println("It's " + _time.currentTime() + " and I'm alive!");
020        _time.enqueue(10+_time.currentTime(), this);
021      }
022    }
