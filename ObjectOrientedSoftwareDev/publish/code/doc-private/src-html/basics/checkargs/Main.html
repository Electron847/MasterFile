001    package basics.checkargs;
002    public class Main {
003      private Main() {}
004      static public void main (final String[] args) {
005        System.out.println(new Person("bob"));
006        System.out.println(new Person(null));
007      }
008    }
009    final class Person {
010      final private String _name;
011      public Person(String name) {
012        if (name == null)
013          throw new IllegalArgumentException();
014        _name = name;
015      }
016      public String toString() { return "Person(" + _name + ")"; };
017    }
