001    package basics.hashmutable;
002    import java.util.Date;
003    import java.util.Hashtable;
004    // example from http://www.vipan.com/htdocs/hashcode_help.html
005    public class Main {
006      public static void main (String[] args) {
007        Hashtable<Date,String> map = new Hashtable<Date,String>();
008        long time = System.currentTimeMillis();
009        Date dt = new Date(time);
010        Date dt2 = new Date(time);
011        
012        System.out.println("dt.toString() = " + dt.toString());
013        System.out.println("dt2.toString() = " + dt2.toString());
014        System.out.println("Is dt2.equals(dt)? = " + dt2.equals(dt));
015        
016        map.put(dt, "blah");
017        System.out.println("map.get(dt) = " + map.get(dt) );
018        System.out.println("map.get(dt2) = " + map.get(dt2) );
019        
020        // Change dt by adding a day to its time
021        long newTime = time + 24*60*60*1000;
022        dt.setTime(newTime);
023        
024        System.out.println("\nAfter dt.setTime(newTime):");
025        System.out.println("dt.toString() = " + dt.toString());
026        System.out.println("dt2.toString() = " + dt2.toString());
027        System.out.println("Is dt2.equals(dt)? = " + dt2.equals(dt));
028        System.out.println("map.get(dt) = " + map.get(dt) );
029        System.out.println("map.get(dt2) = " + map.get(dt2) );
030        
031        System.out.println("\nmap = " + map.toString() );
032      }
033    }
