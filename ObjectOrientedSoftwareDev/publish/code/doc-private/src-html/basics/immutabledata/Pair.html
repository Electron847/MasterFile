001    package basics.immutabledata;
002    final public class Pair<S extends Comparable<S>, T extends Comparable<T>>
003      implements Comparable<Pair<S,T>>
004    {
005      final private S _x;
006      final private T _y;
007      public Pair(S x, T y) {
008        if (x == null || y == null)
009          throw new IllegalArgumentException();
010        _x = x;
011        _y = y;
012      }
013      public S first() { return _x; }
014      public T second() { return _y; }
015      public String toString() { return "Pair(" + _x + "," + _y + ")"; }
016      public boolean equals(Object thatObject) {
017        if (this == thatObject)
018          return true;
019        // equals should not throw ClassCastException
020        if (!(this.getClass().equals(thatObject.getClass())))
021          return false;
022        Pair that = (Pair) thatObject;
023        return _x.equals(that._x)
024            && _y.equals(that._y);
025      }
026      private int hcode;
027      public int hashCode() {
028        if (hcode == 0) {
029          hcode = 17;
030          hcode = 37*hcode + _x.hashCode();
031          hcode = 37*hcode + _y.hashCode();
032        }
033        return hcode;
034      }
035      public int compareTo(Pair<S,T> that) {
036        // compareTo may throw ClassCastException
037        int ix = _x.compareTo(that._x);
038        if (ix != 0)
039          return ix;
040        return _y.compareTo(that._y);
041      }
042    }
