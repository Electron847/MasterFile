011    
012    abstract class Print {
013      int py;
014      Print(int y) { py = y; }
015      abstract void print();
016    }
017    
018    class C {
019      int cx;
020      C(int x) { cx = x; }
021      void f() {
022        Print p = new Print(27) {
023            public void print() {
024              System.out.println(" cx=" + Integer.toString(cx) +
025                                 " py=" + Integer.toString(py));
026            }
027          };
028        p.print();
029      }
030    }
