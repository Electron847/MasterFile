001    package basics.mutablebasic;
002    import basics.immutabledata.Pair;
003    public class Main {
004      private Main() {}
005      static public void main (final String[] args) {
006        final PairBuilder<Integer,String> pb1 = new PairBuilder<Integer,String>();
007        pb1.setFirst(42);
008        pb1.setSecond("dog");
009        System.out.println(pb1);
010        final Pair<Integer,String> p1 = pb1.toPair();
011        System.out.println(p1);
012        
013        final PairBuilder<Integer,String> pb2 = new PairBuilder<Integer,String>();
014        final Pair<Integer,String> p2 = pb2.toPair();
015      }
016    }
017    final class PairBuilder<S extends Comparable<S>, T extends Comparable<T>> {
018      private S _x;
019      private T _y;
020      public PairBuilder() { }
021      public void setFirst(S x) { _x = x; }
022      public void setSecond(T y) { _y = y; }
023      public Pair<S,T> toPair() {
024        if (_x == null || _y == null)
025          throw new NullPointerException();
026        return new Pair<S,T>(_x,_y);
027      }
028    }
