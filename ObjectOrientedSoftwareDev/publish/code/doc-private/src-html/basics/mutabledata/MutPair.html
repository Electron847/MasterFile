001    package basics.mutabledata;
002    import basics.immutabledata.Pair;
003    // Do not override equals or hashCode for mutable data.
004    // Do not make mutable data Comparable.
005    final public class MutPair<S,T>
006      implements Cloneable
007    {
008      private S _x;
009      private T _y;
010      public MutPair() { }
011      public MutPair(S x, T y) { _x = x; _y = y; }
012      public void setFirst(S x) { _x = x; }
013      public void setSecond(T y) { _y = y; }
014      public S first() { return _x; }
015      public T second() { return _y; }
016      public String toString() { return "MutPair(" + _x + "," + _y + ")"; }
017      // Clone does not play nice with generics because it returns Object.
018      // Deep clone is difficult to implement generically because:
019      // (1) cast to MutPair is allowed, not to MutPair<S,T>; we need the
020      //     latter in order to get at the fields.
021      // (2) Cloneable type does not have public clone method, so we
022      //     cannot call _x.clone() or _y.clone(); this would be true even
023      //     if we had "MutPair<S extends Cloneable,T extends Cloneable>"
024      public Object clone() {
025        try {
026          return super.clone();
027        } catch (CloneNotSupportedException e) {
028          throw new InternalError();
029        }
030      }
031    }
