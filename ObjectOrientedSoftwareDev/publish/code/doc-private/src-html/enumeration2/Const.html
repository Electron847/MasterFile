001    package enumeration2;
002    
003    public interface Expr {
004      void printPostorder();
005      int evaluate();
006    }
007    
008    class Const implements Expr {
009      private final int _v;
010      public Const(int v) {
011        _v = v;
012      }
013    
014      public int evaluate() {
015        return _v;
016      }
017      public void printPostorder() {
018        System.out.print(_v + " ");
019      }
020    }
021    
022    class BinOp implements Expr {
023      private final Expr _l;
024      private final Expr _r;
025      private final Op _op;
026    
027      public BinOp(Expr l, Op op, Expr r) {
028        if ((l == null) || (op == null) || (r == null)) {
029          throw new IllegalArgumentException();
030        }
031        _op = op;
032        _l = l;
033        _r = r;
034      }
035    
036      public int evaluate() {
037        return _op.eval(_l.evaluate(), _r.evaluate());
038      }
039      public void printPostorder() {
040        _l.printPostorder();
041        _r.printPostorder();
042        System.out.print(_op + " ");
043      }
044    }
