001    package headfirst.command.remote;
002    
003    import java.util.*;
004    
005    //
006    // This is the invoker
007    //
008    public class RemoteControl {
009            Command[] onCommands;
010            Command[] offCommands;
011     
012            public RemoteControl() {
013                    onCommands = new Command[7];
014                    offCommands = new Command[7];
015     
016                    Command noCommand = new NoCommand();
017                    for (int i = 0; i < 7; i++) {
018                            onCommands[i] = noCommand;
019                            offCommands[i] = noCommand;
020                    }
021            }
022      
023            public void setCommand(int slot, Command onCommand, Command offCommand) {
024                    onCommands[slot] = onCommand;
025                    offCommands[slot] = offCommand;
026            }
027     
028            public void onButtonWasPushed(int slot) {
029                    onCommands[slot].execute();
030            }
031     
032            public void offButtonWasPushed(int slot) {
033                    offCommands[slot].execute();
034            }
035      
036            public String toString() {
037                    StringBuilder stringBuff = new StringBuilder();
038                    stringBuff.append("\n------ Remote Control -------\n");
039                    for (int i = 0; i < onCommands.length; i++) {
040                            stringBuff.append("[slot " + i + "] " + onCommands[i].getClass().getName()
041                                    + "    " + offCommands[i].getClass().getName() + "\n");
042                    }
043                    return stringBuff.toString();
044            }
045    }
