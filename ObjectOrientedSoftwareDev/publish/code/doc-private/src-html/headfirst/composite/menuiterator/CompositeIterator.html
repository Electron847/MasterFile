001    package headfirst.composite.menuiterator;
002    
003     
004    import java.util.*;
005      
006    public class CompositeIterator implements Iterator {
007            Stack<Object> stack = new Stack<Object>();
008       
009            public CompositeIterator(Iterator iterator) {
010                    stack.push(iterator);
011            }
012       
013            public Object next() {
014                    if (hasNext()) {
015                            Iterator iterator = (Iterator) stack.peek();
016                            MenuComponent component = (MenuComponent) iterator.next();
017                            if (component instanceof Menu) {
018                                    stack.push(component.createIterator());
019                            } 
020                            return component;
021                    } else {
022                            return null;
023                    }
024            }
025      
026            public boolean hasNext() {
027                    if (stack.empty()) {
028                            return false;
029                    } else {
030                            Iterator iterator = (Iterator) stack.peek();
031                            if (!iterator.hasNext()) {
032                                    stack.pop();
033                                    return hasNext();
034                            } else {
035                                    return true;
036                            }
037                    }
038            }
039       
040            public void remove() {
041                    throw new UnsupportedOperationException();
042            }
043    }
