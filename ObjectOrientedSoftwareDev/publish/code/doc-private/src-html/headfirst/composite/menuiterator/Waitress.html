001    package headfirst.composite.menuiterator;
002    
003    import java.util.Iterator;
004      
005    public class Waitress {
006            MenuComponent allMenus;
007     
008            public Waitress(MenuComponent allMenus) {
009                    this.allMenus = allMenus;
010            }
011     
012            public void printMenu() {
013                    allMenus.print();
014            }
015      
016            public void printVegetarianMenu() {
017                    Iterator iterator = allMenus.createIterator();
018    
019                    System.out.println("\nVEGETARIAN MENU\n----");
020                    while (iterator.hasNext()) {
021                            MenuComponent menuComponent = 
022                                            (MenuComponent)iterator.next();
023                            try {
024                                    if (menuComponent.isVegetarian()) {
025                                            menuComponent.print();
026                                    }
027                            } catch (UnsupportedOperationException e) {}
028                    }
029            }
030    }
