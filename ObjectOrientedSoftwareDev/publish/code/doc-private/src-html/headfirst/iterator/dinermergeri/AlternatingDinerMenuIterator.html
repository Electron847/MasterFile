001    package headfirst.iterator.dinermergeri;
002    
003    import java.util.Iterator;
004    import java.util.Calendar;
005    
006    public class AlternatingDinerMenuIterator implements Iterator {
007            MenuItem[] items;
008            int position;
009    
010            public AlternatingDinerMenuIterator(MenuItem[] items) {
011                    this.items = items;
012                    Calendar rightNow = Calendar.getInstance();
013                    position = rightNow.DAY_OF_WEEK % 2;
014            }
015            public Object next() {
016                    MenuItem menuItem = items[position];
017                    position = position + 2;
018                    return menuItem;
019            }
020            public boolean hasNext() {
021                    if (position >= items.length || items[position] == null) {
022                            return false;
023                    } else {
024                            return true;
025                    }
026            }
027            public void remove() {
028                    throw new UnsupportedOperationException(
029                            "Alternating Diner Menu Iterator does not support remove()");
030            }
031    }
