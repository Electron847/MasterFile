001    package headfirst.iterator.dinermergeri;
002    
003    import java.util.Iterator;
004    
005    public class DinerMenu implements Menu {
006            static final int MAX_ITEMS = 6;
007            int numberOfItems = 0;
008            MenuItem[] menuItems;
009      
010            public DinerMenu() {
011                    menuItems = new MenuItem[MAX_ITEMS];
012     
013                    addItem("Vegetarian BLT",
014                            "(Fakin') Bacon with lettuce & tomato on whole wheat", true, 2.99);
015                    addItem("BLT",
016                            "Bacon with lettuce & tomato on whole wheat", false, 2.99);
017                    addItem("Soup of the day",
018                            "Soup of the day, with a side of potato salad", false, 3.29);
019                    addItem("Hotdog",
020                            "A hot dog, with saurkraut, relish, onions, topped with cheese",
021                            false, 3.05);
022                    addItem("Steamed Veggies and Brown Rice",
023                            "Steamed vegetables over brown rice", true, 3.99);
024                    addItem("Pasta",
025                            "Spaghetti with Marinara Sauce, and a slice of sourdough bread",
026                            true, 3.89);
027            }
028      
029            public void addItem(String name, String description, 
030                                 boolean vegetarian, double price) 
031            {
032                    MenuItem menuItem = new MenuItem(name, description, vegetarian, price);
033                    if (numberOfItems >= MAX_ITEMS) {
034                            System.err.println("Sorry, menu is full!  Can't add item to menu");
035                    } else {
036                            menuItems[numberOfItems] = menuItem;
037                            numberOfItems = numberOfItems + 1;
038                    }
039            }
040     
041            public MenuItem[] getMenuItems() {
042                    return menuItems;
043            }
044      
045            public Iterator createIterator() {
046                    return new DinerMenuIterator(menuItems);
047                    //return new AlternatingDinerMenuIterator(menuItems);
048            }
049     
050            // other menu methods here
051    }
