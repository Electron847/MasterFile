001    package headfirst.observer.WeatherStation;
002    
003    import java.util.*;
004    
005    public class ForecastDisplay implements Observer, DisplayElement {
006            private float currentPressure = 29.92f;  
007            private float lastPressure;
008            private WeatherData weatherData;
009    
010            public ForecastDisplay(WeatherData weatherData) {
011                    this.weatherData = weatherData;
012                    weatherData.registerObserver(this);
013            }
014    
015            public void update(float temp, float humidity, float pressure) {
016                    lastPressure = currentPressure;
017                    currentPressure = pressure;
018    
019                    display();
020            }
021    
022            public void display() {
023                    System.out.print("Forecast: ");
024                    if (currentPressure > lastPressure) {
025                            System.out.println("Improving weather on the way!");
026                    } else if (currentPressure == lastPressure) {
027                            System.out.println("More of the same");
028                    } else if (currentPressure < lastPressure) {
029                            System.out.println("Watch out for cooler, rainy weather");
030                    }
031            }
032    }
