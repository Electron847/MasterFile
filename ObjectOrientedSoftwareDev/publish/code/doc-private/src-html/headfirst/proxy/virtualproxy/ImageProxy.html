001    package headfirst.proxy.virtualproxy;
002    
003    import java.net.*;
004    import java.awt.*;
005    import java.awt.event.*;
006    import javax.swing.*;
007    
008    class ImageProxy implements Icon {
009            ImageIcon imageIcon;
010            URL imageURL;
011            Thread retrievalThread;
012            boolean retrieving = false;
013         
014            public ImageProxy(URL url) { imageURL = url; }
015         
016            public int getIconWidth() {
017                    if (imageIcon != null) {
018                return imageIcon.getIconWidth();
019            } else {
020                            return 800;
021                    }
022            }
023     
024            public int getIconHeight() {
025                    if (imageIcon != null) {
026                return imageIcon.getIconHeight();
027            } else {
028                            return 600;
029                    }
030            }
031         
032            public void paintIcon(final Component c, Graphics  g, int x,  int y) {
033                    if (imageIcon != null) {
034                            imageIcon.paintIcon(c, g, x, y);
035                    } else {
036                            g.drawString("Loading CD cover, please wait...", x+300, y+190);
037                            if (!retrieving) {
038                                    retrieving = true;
039    
040                                    retrievalThread = new Thread(new Runnable() {
041                                            public void run() {
042                                                    try {
043                                                            imageIcon = new ImageIcon(imageURL, "CD Cover");
044                                                            c.repaint();
045                                                    } catch (Exception e) {
046                                                            e.printStackTrace();
047                                                    }
048                                            }
049                                    });
050                                    retrievalThread.start();
051                            }
052                    }
053            }
054    }
