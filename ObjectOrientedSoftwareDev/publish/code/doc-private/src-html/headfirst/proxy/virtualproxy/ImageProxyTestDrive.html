001    package headfirst.proxy.virtualproxy;
002    
003    import java.net.*;
004    import java.awt.*;
005    import java.awt.event.*;
006    import javax.swing.*;
007    import java.util.*;
008    
009    public class ImageProxyTestDrive {
010            ImageComponent imageComponent;
011            JFrame frame = new JFrame("CD Cover Viewer");
012      JMenuBar menuBar;
013      JMenu menu;
014            Hashtable<String,String> cds = new Hashtable<String,String>();
015      
016            public static void main (String[] args) throws Exception {
017                    ImageProxyTestDrive testDrive = new ImageProxyTestDrive();
018            }
019      
020            public ImageProxyTestDrive() throws Exception{
021                    cds.put("Ambient: Music for Airports","http://images.amazon.com/images/P/B000003S2K.01.LZZZZZZZ.jpg");
022        cds.put("Buddha Bar","http://images.amazon.com/images/P/B00009XBYK.01.LZZZZZZZ.jpg");
023        cds.put("Ima","http://images.amazon.com/images/P/B000005IRM.01.LZZZZZZZ.jpg");
024        cds.put("Karma","http://images.amazon.com/images/P/B000005DCB.01.LZZZZZZZ.gif");
025        cds.put("MCMXC A.D.","http://images.amazon.com/images/P/B000002URV.01.LZZZZZZZ.jpg");
026        cds.put("Northern Exposure","http://images.amazon.com/images/P/B000003SFN.01.LZZZZZZZ.jpg");
027        cds.put("Selected Ambient Works, Vol. 2","http://images.amazon.com/images/P/B000002MNZ.01.LZZZZZZZ.jpg");
028        
029                    URL initialURL = new URL((String)cds.get("Selected Ambient Works, Vol. 2"));
030                    menuBar = new JMenuBar();
031                    menu = new JMenu("Favorite CDs");
032        menuBar.add(menu);
033        frame.setJMenuBar(menuBar);
034        
035                    for(Enumeration e = cds.keys(); e.hasMoreElements();) {
036                            String name = (String)e.nextElement();
037          JMenuItem menuItem = new JMenuItem(name);
038          menu.add(menuItem); 
039          menuItem.addActionListener(new ActionListener() {
040              public void actionPerformed(ActionEvent event) {
041                imageComponent.setIcon(new ImageProxy(getCDUrl(event.getActionCommand())));
042                frame.repaint();
043              }
044            });
045                    }
046                    
047                    // set up frame and menus
048        
049                    Icon icon = new ImageProxy(initialURL);
050                    imageComponent = new ImageComponent(icon);
051                    frame.getContentPane().add(imageComponent);
052                    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
053                    frame.setSize(800,600);
054                    frame.setVisible(true);
055        
056            }
057      
058            URL getCDUrl(String name) {
059                    try {
060                            return new URL((String)cds.get(name));
061                    } catch (MalformedURLException e) {
062                            e.printStackTrace();
063                            return null;
064                    }
065            }
066    }
