001    package headfirst.state.gumballstate;
002    
003    import java.util.Random;
004    
005    public class HasQuarterState implements State {
006            GumballMachine gumballMachine;
007     
008            public HasQuarterState(GumballMachine gumballMachine) {
009                    this.gumballMachine = gumballMachine;
010            }
011      
012            public void insertQuarter() {
013                    System.out.println("You can't insert another quarter");
014            }
015     
016            public void ejectQuarter() {
017                    System.out.println("Quarter returned");
018                    gumballMachine.setState(gumballMachine.getNoQuarterState());
019            }
020     
021            public void turnCrank() {
022                    System.out.println("You turned...");
023                    gumballMachine.setState(gumballMachine.getSoldState());
024            }
025    
026        public void dispense() {
027            System.out.println("No gumball dispensed");
028        }
029     
030            public String toString() {
031                    return "waiting for turn of crank";
032            }
033    }
