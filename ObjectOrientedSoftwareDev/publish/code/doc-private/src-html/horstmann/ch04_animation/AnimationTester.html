001    package horstmann.ch04_animation;
002    import java.awt.*;
003    import java.awt.event.*;
004    import javax.swing.*;
005    
006    /**
007       This program implements an animation that moves
008       a car shape.
009    */
010    public class AnimationTester
011    {
012       public static void main(String[] args)
013       {
014          JFrame frame = new JFrame();
015    
016          final MoveableShape shape
017                = new CarShape(0, 0, CAR_WIDTH);
018    
019          ShapeIcon icon = new ShapeIcon(shape,
020                ICON_WIDTH, ICON_HEIGHT);
021    
022          final JLabel label = new JLabel(icon);
023          frame.setLayout(new FlowLayout());
024          frame.add(label);
025    
026          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
027          frame.pack();
028          frame.setVisible(true);
029    
030          final int DELAY = 100;
031          // Milliseconds between timer ticks
032          Timer t = new Timer(DELAY, new
033             ActionListener()
034             {
035                public void actionPerformed(ActionEvent event)
036                {
037                   shape.translate(1, 0);
038                   label.repaint();
039                }
040             });
041          t.start();
042       }
043    
044       private static final int ICON_WIDTH = 400;
045       private static final int ICON_HEIGHT = 100;
046       private static final int CAR_WIDTH = 100;
047    }
