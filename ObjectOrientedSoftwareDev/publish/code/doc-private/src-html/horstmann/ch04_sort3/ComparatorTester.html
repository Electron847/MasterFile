001    package horstmann.ch04_sort3;
002    import java.util.*;
003    
004    public class ComparatorTester
005    {
006       public static void main(String[] args)
007       {
008          ArrayList<Country> countries = new ArrayList<Country>();
009          countries.add(new Country("Uruguay", 176220));
010          countries.add(new Country("Thailand", 514000));
011          countries.add(new Country("Belgium", 30510));
012    
013          Collections.sort(countries, new
014             Comparator<Country>()
015             {
016                public int compare(Country country1, Country country2)
017                {
018                   return country1.getName()
019                      .compareTo(country2.getName());
020                }
021             });
022    
023          // Now the array list is sorted by name
024          for (Country c : countries)
025             System.out.println(c.getName() + " " + c.getArea());
026       }
027    }
