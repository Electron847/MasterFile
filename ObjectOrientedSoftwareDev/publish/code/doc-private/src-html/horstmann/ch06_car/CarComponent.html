001    package horstmann.ch06_car;
002    import java.awt.*;
003    import java.awt.event.*;
004    import java.awt.geom.*;
005    import javax.swing.*;
006    import java.util.*;
007    
008    /**
009       A component that shows a scene composed of items.
010    */
011    public class CarComponent extends JComponent
012    {
013       public CarComponent()
014       {
015          car = new CarShape(20, 20, 50);
016          addMouseListener(new
017             MouseAdapter()
018             {
019                public void mousePressed(MouseEvent event)
020                {
021                   mousePoint = event.getPoint();
022                   if (!car.contains(mousePoint))
023                      mousePoint = null;
024                }
025             });
026    
027          addMouseMotionListener(new
028             MouseMotionAdapter()
029             {
030                public void mouseDragged(MouseEvent event)
031                {
032                   if (mousePoint == null) return;
033                   Point lastMousePoint = mousePoint;
034                   mousePoint = event.getPoint();
035    
036                   double dx = mousePoint.getX() - lastMousePoint.getX();
037                   double dy = mousePoint.getY() - lastMousePoint.getY();
038                   car.translate((int) dx, (int) dy);
039                   repaint();
040                }
041             });
042       }
043    
044       public void paintComponent(Graphics g)
045       {
046          Graphics2D g2 = (Graphics2D) g;
047          car.draw(g2);
048       }
049    
050       private CarShape car;
051       private Point mousePoint;
052    }
