001    package horstmann.ch06_scene1;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       A shape that manages its selection state.
007    */
008    public abstract class SelectableShape implements SceneShape
009    {
010       public void setSelected(boolean b)
011       {
012          selected = b;
013       }
014    
015       public boolean isSelected()
016       {
017          return selected;
018       }
019    
020       private boolean selected;
021    }
