001    package horstmann.ch06_scene2;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       A house shape.
007    */
008    public class HouseShape extends SelectableShape
009    {
010       /**
011          Constructs a house shape.
012          @param x the left of the bounding rectangle
013          @param y the top of the bounding rectangle
014          @param width the width of the bounding rectangle
015       */
016       public HouseShape(int x, int y, int width)
017       {
018          this.x = x;
019          this.y = y;
020          this.width = width;
021       }
022    
023       public void draw(Graphics2D g2)
024       {
025          Rectangle2D.Double base 
026             = new Rectangle2D.Double(x, y + width, width, width);
027    
028          // The left bottom of the roof
029          Point2D.Double r1
030             = new Point2D.Double(x, y + width);
031          // The top of the roof
032          Point2D.Double r2
033             = new Point2D.Double(x + width / 2, y);
034          // The right bottom of the roof
035          Point2D.Double r3
036             = new Point2D.Double(x + width, y + width);
037    
038          Line2D.Double roofLeft
039             = new Line2D.Double(r1, r2);
040          Line2D.Double roofRight
041             = new Line2D.Double(r2, r3);
042    
043          g2.draw(base);
044          g2.draw(roofLeft);
045          g2.draw(roofRight);
046       }
047       
048       public boolean contains(Point2D p)
049       {
050          return x <= p.getX() && p.getX() <= x + width 
051             && y <= p.getY() && p.getY() <= y + 2 * width;
052       }
053    
054       public void translate(int dx, int dy)
055       {
056          x += dx;
057          y += dy;
058       }
059    
060       private int x;
061       private int y;
062       private int width;
063    }
