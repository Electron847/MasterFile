001    package horstmann.ch07_reflect1;
002    import java.lang.reflect.*;
003    import java.io.*;
004    
005    /**
006       This program prints "Hello, World" the hard way,
007       using reflection.
008    */
009    public class HardHello
010    {
011       public static void main(String[] args)
012          throws NoSuchMethodException, IllegalAccessException,
013                 InvocationTargetException
014       {
015          Method m = PrintStream.class.getDeclaredMethod(
016                "println", String.class);
017          m.invoke(System.out, "Hello, World!");
018       }
019    }
