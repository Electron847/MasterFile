001    package horstmann.ch07_reflect2;
002    import java.lang.reflect.*;
003    import java.util.*;
004    
005    /**
006       This program shows how to use reflection to print
007       the names and values of all nonstatic fields of an object.
008    */
009    public class FieldTester
010    {
011       public static void main(String[] args)
012             throws IllegalAccessException
013       {
014          Random r = new Random();
015          System.out.print(spyFields(r));
016          r.nextInt();
017          System.out.println("\nAfter calling nextInt:\n");      
018          System.out.print(spyFields(r));
019       }
020    
021       /**
022          Spies on the field names and values of an object.
023          @param obj the object whose fields to format
024          @return a string containing the names and values of
025          all fields of obj
026       */
027       public static String spyFields(Object obj)
028             throws IllegalAccessException
029       {
030          StringBuilder buffer = new StringBuilder();
031          Field[] fields = obj.getClass().getDeclaredFields();
032          for (Field f : fields)
033          {
034             if (!Modifier.isStatic(f.getModifiers()))
035             {
036                f.setAccessible(true);
037                Object value = f.get(obj);
038                buffer.append(f.getType().getName());
039                buffer.append(" ");
040                buffer.append(f.getName());
041                buffer.append("=");
042                buffer.append("" + value);
043                buffer.append("\n");
044             }
045          }
046          return buffer.toString();
047       }
048    }
