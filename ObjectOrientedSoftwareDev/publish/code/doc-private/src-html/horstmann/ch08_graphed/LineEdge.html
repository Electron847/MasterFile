001    package horstmann.ch08_graphed;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       An edge that is shaped like a straight line.
007    */
008    public class LineEdge extends AbstractEdge
009    {
010       public void draw(Graphics2D g2)
011       {
012          g2.draw(getConnectionPoints());
013       }
014    
015       public boolean contains(Point2D aPoint)
016       {
017          final double MAX_DIST = 2;
018          return getConnectionPoints().ptSegDist(aPoint) 
019             < MAX_DIST;
020       }
021    }
