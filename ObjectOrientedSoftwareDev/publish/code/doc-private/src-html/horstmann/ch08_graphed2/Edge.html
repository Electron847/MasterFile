001    package horstmann.ch08_graphed2;
002    import java.awt.*;
003    import java.awt.geom.*;
004    import java.io.*;
005    
006    /**
007       An edge in a graph.
008    */
009    public interface Edge extends Serializable, Cloneable
010    {
011       /**
012          Draw the edge.
013          @param g2 the graphics context
014       */
015       void draw(Graphics2D g2);
016    
017       /**
018          Tests whether the edge contains a point.
019          @param aPoint the point to test
020          @return true if this edge contains aPoint
021       */
022       boolean contains(Point2D aPoint);
023    
024       /**
025          Connects this edge to two nodes.
026          @param aStart the starting node
027          @param anEnd the ending node
028       */
029       void connect(Node aStart, Node anEnd);
030    
031       /**
032          Gets the starting node.
033          @return the starting node
034       */
035       Node getStart();
036    
037       /**
038          Gets the ending node.
039          @return the ending node
040       */
041       Node getEnd();
042    
043       /**
044          Gets the points at which this edge is connected to
045          its nodes.
046          @return a line joining the two connection points
047       */
048       Line2D getConnectionPoints();
049    
050       /**
051          Gets the smallest rectangle that bounds this edge.
052          The bounding rectangle contains all labels.
053          @return the bounding rectangle
054       */
055       Rectangle2D getBounds(Graphics2D g2);
056    
057       Object clone();
058    }
