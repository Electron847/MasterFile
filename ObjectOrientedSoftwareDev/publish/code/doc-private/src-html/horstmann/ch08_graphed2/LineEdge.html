001    package horstmann.ch08_graphed2;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       An edge that is shaped like a straight line.
007    */
008    public class LineEdge extends AbstractEdge
009    {
010       public LineEdge()
011       {
012          lineStyle = LineStyle.SOLID;
013       }
014    
015       public void draw(Graphics2D g2)
016       {
017          Stroke oldStroke = g2.getStroke();
018          g2.setStroke(lineStyle.getStroke());
019          g2.draw(getConnectionPoints());
020          g2.setStroke(oldStroke);
021       }
022    
023       public boolean contains(Point2D aPoint)
024       {
025          final double MAX_DIST = 2;
026          return getConnectionPoints().ptSegDist(aPoint) 
027             < MAX_DIST;
028       }
029    
030       /**
031          Sets the line style property.
032          @param newValue the new value
033       */
034       public void setLineStyle(LineStyle newValue) { lineStyle = newValue; }
035    
036       /**
037          Gets the line style property.
038          @return the line style
039       */
040       public LineStyle getLineStyle() { return lineStyle; }
041       
042       private LineStyle lineStyle;
043    }
