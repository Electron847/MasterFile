001    package horstmann.ch08_umleditor;
002    import java.awt.geom.*;
003    
004    /**
005       A UML class diagram.
006    */
007    public class ClassDiagramGraph extends Graph
008    {
009    
010       public boolean connect(Edge e, Point2D p1, Point2D p2)
011       {
012          Node n1 = findNode(p1);
013          Node n2 = findNode(p2);
014          if (n1 == n2) return false;
015          return super.connect(e, p1, p2);
016       }
017    
018       public Node[] getNodePrototypes()
019       {
020          return new Node[] { new ClassNode() };
021       }
022    
023       public Edge[] getEdgePrototypes()
024       {
025          ClassRelationshipEdge[] e = new ClassRelationshipEdge[6];
026    
027          e[0] = new ClassRelationshipEdge();
028          e[0].setLineStyle(LineStyle.DOTTED);
029          e[0].setEndArrowHead(ArrowHead.V);
030    
031          e[1] = new ClassRelationshipEdge();
032          e[1].setBentStyle(BentStyle.VHV);
033          e[1].setEndArrowHead(ArrowHead.TRIANGLE);
034    
035          e[2] = new ClassRelationshipEdge();
036          e[2].setBentStyle(BentStyle.VHV);
037          e[2].setLineStyle(LineStyle.DOTTED);
038          e[2].setEndArrowHead(ArrowHead.TRIANGLE);
039    
040          e[3] = new ClassRelationshipEdge();
041          e[3].setBentStyle(BentStyle.HVH);
042          e[3].setEndArrowHead(ArrowHead.V);
043    
044          e[4] = new ClassRelationshipEdge();
045          e[4].setBentStyle(BentStyle.HVH);
046          e[4].setStartArrowHead(ArrowHead.DIAMOND);
047    
048          e[5] = new ClassRelationshipEdge();
049          e[5].setBentStyle(BentStyle.HVH);
050          e[5].setStartArrowHead(ArrowHead.BLACK_DIAMOND);
051    
052          return e;
053      }
054    }
