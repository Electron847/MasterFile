001    package horstmann.ch08_umleditor;
002    import java.awt.Dimension;
003    import java.awt.Graphics2D;
004    import java.awt.geom.Rectangle2D;
005    import java.io.Serializable;
006    import java.util.StringTokenizer;
007    
008    import javax.swing.JLabel;
009    
010    /**
011       A string that can extend over multiple lines.
012    */
013    public class MultiLineString implements Cloneable, Serializable
014    {
015       /**
016          Constructs an empty, centered, normal size multiline
017          string that is not underlined.
018       */
019       public MultiLineString() 
020       { 
021          text = ""; 
022          justification = CENTER;
023          size = NORMAL;
024          underlined = false;    
025       }
026       /**
027          Sets the value of the text property.
028          @param newValue the text of the multiline string
029       */
030       public void setText(String newValue) { text = newValue; }
031       /**
032          Gets the value of the text property.
033          @return the text of the multiline string
034       */
035       public String getText() { return text; }
036       /**
037          Sets the value of the justification property.
038          @param newValue the justification, one of LEFT, CENTER, 
039          RIGHT
040       */
041       public void setJustification(int newValue) { justification = newValue; }
042       /**
043          Gets the value of the justification property.
044          @return the justification, one of LEFT, CENTER, 
045          RIGHT
046       */
047       public int getJustification() { return justification; }
048       /**
049          Gets the value of the underlined property.
050          @return true if the text is underlined
051       */
052       public boolean isUnderlined() { return underlined; }
053       /**
054          Sets the value of the underlined property.
055          @param newValue true to underline the text
056       */
057       public void setUnderlined(boolean newValue) { underlined = newValue; }
058       /**
059          Sets the value of the size property.
060          @param newValue the size, one of SMALL, NORMAL, LARGE
061       */
062       public void setSize(int newValue) { size = newValue; }
063       /**
064          Gets the value of the size property.
065          @return the size, one of SMALL, NORMAL, LARGE
066       */
067       public int getSize() { return size; }
068       
069       public String toString()
070       {
071          return text.replace('\n', '|');
072       }
073    
074       private void setLabelText()
075       {
076          StringBuilder prefix = new StringBuilder();
077          StringBuilder suffix = new StringBuilder();
078          StringBuilder htmlText = new StringBuilder();
079          prefix.append("&nbsp;");
080          suffix.insert(0, "&nbsp;");
081          if (underlined) 
082          {
083             prefix.append("<u>");
084             suffix.insert(0, "</u>");
085          }
086          if (size == LARGE)
087          {
088             prefix.append("<font size=\"+1\">");
089             suffix.insert(0, "</font>");
090          }
091          if (size == SMALL)
092          {
093             prefix.append("<font size=\"-1\">");
094             suffix.insert(0, "</font>");
095          }
096          htmlText.append("<html>");
097          StringTokenizer tokenizer = new StringTokenizer(text, "\n");
098          boolean first = true;
099          while (tokenizer.hasMoreTokens())
100          {
101             if (first) first = false; else htmlText.append("<br>");
102             htmlText.append(prefix);
103             htmlText.append(tokenizer.nextToken());
104             htmlText.append(suffix);
105          }      
106          htmlText.append("</html>");
107          label.setText(htmlText.toString());
108          if (justification == LEFT) label.setHorizontalAlignment(JLabel.LEFT);
109          else if (justification == CENTER) label.setHorizontalAlignment(JLabel.CENTER);
110          else if (justification == RIGHT) label.setHorizontalAlignment(JLabel.RIGHT);
111       }
112       
113       /**
114          Gets the bounding rectangle for this multiline string.
115          @param g2 the graphics context
116          @return the bounding rectangle (with top left corner (0,0))
117       */
118       public Rectangle2D getBounds(Graphics2D g2)
119       {
120          if (text.length() == 0) return new Rectangle2D.Double();
121          setLabelText();
122          Dimension dim = label.getPreferredSize();       
123          return new Rectangle2D.Double(0, 0, dim.getWidth(), dim.getHeight());
124       }
125    
126       /**
127          Draws this multiline string inside a given rectangle
128          @param g2 the graphics context
129          @param r the rectangle into which to place this multiline string
130       */
131       public void draw(Graphics2D g2, Rectangle2D r)
132       {
133          setLabelText();
134          label.setFont(g2.getFont());
135          label.setBounds(0, 0, (int) r.getWidth(), (int) r.getHeight());
136          g2.translate(r.getX(), r.getY());
137          label.paint(g2);
138          g2.translate(-r.getX(), -r.getY());        
139       }
140    
141       public Object clone()
142       {
143          try
144          {
145             return super.clone();
146          }
147          catch (CloneNotSupportedException exception)
148          {
149             return null;
150          }
151       }
152    
153       public static final int LEFT = 0;
154       public static final int CENTER = 1;
155       public static final int RIGHT = 2;
156       public static final int LARGE = 3;
157       public static final int NORMAL = 4;
158       public static final int SMALL = 5;
159    
160       private static final int GAP = 3;
161    
162       private String text;
163       private int justification;
164       private int size;
165       private boolean underlined;
166       
167       private static JLabel label = new JLabel();
168    }
