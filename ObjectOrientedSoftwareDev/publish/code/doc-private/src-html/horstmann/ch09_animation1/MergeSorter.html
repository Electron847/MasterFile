001    package horstmann.ch09_animation1;
002    import java.util.*;
003    
004    /**
005       This class carries out the merge sort algorithm.
006    */
007    public class MergeSorter
008    {
009       /**
010          Sorts an array, using the merge sort algorithm.
011          @param a the array to sort
012          @param comp the comparator to compare array elements
013       */
014       public static <E> void sort(E[] a, Comparator<? super E> comp)
015       {
016          mergeSort(a, 0, a.length - 1, comp);
017       }
018    
019       /**
020          Sorts a range of an array, using the merge sort
021          algorithm.
022          @param a the array to sort
023          @param from the first index of the range to sort
024          @param to the last index of the range to sort
025          @param comp the comparator to compare array elements
026       */
027       private static <E> void mergeSort(E[] a, int from, int to,
028          Comparator<? super E> comp)
029       {
030          if (from == to) return;
031          int mid = (from + to) / 2;
032          // Sort the first and the second half
033          mergeSort(a, from, mid, comp);
034          mergeSort(a, mid + 1, to, comp);
035          merge(a, from, mid, to, comp);
036       }
037    
038       /**
039          Merges two adjacent subranges of an array
040          @param a the array with entries to be merged
041          @param from the index of the first element of the
042             first range
043          @param mid the index of the last element of the
044             first range
045          @param to the index of the last element of the
046             second range
047          @param comp the comparator to compare array elements
048       */
049       private static <E> void merge(E[] a,
050          int from, int mid, int to, Comparator<? super E> comp)
051       {
052          int n = to - from + 1;
053             // Size of the range to be merged
054    
055          // Merge both halves into a temporary array b
056          ArrayList<E> b = new ArrayList<E>(n);
057    
058          int i1 = from;
059             // Next element to consider in the first range
060          int i2 = mid + 1;
061             // Next element to consider in the second range
062          int j = 0;
063             // Next open position in b
064    
065          // As long as neither i1 nor i2 past the end, move
066          // the smaller element into b
067          while (i1 <= mid && i2 <= to)
068          {
069             if (comp.compare(a[i1], a[i2]) < 0)
070             {
071                b.set(j,a[i1]);
072                i1++;
073             }
074             else
075             {
076                b.set(j,a[i2]);
077                i2++;
078             }
079             j++;
080          }
081    
082          // Note that only one of the two while loops
083          // below is executed
084    
085          // Copy any remaining entries of the first half
086          while (i1 <= mid)
087          {
088             b.set(j,a[i1]);
089             i1++;
090             j++;
091          }
092    
093          // Copy any remaining entries of the second half
094          while (i2 <= to)
095          {
096             b.set(j,a[i2]);
097             i2++;
098             j++;
099          }
100    
101          // Copy back from the temporary array
102          for (j = 0; j < n; j++)
103            a[from + j] = b.get(j);
104       }
105    }
