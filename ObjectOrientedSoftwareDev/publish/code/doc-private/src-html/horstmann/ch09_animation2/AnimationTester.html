001    package horstmann.ch09_animation2;
002    import java.awt.*;
003    import java.awt.event.*;
004    import javax.swing.*;
005    import java.util.concurrent.*;
006    
007    /**
008       This program animates a sort algorithm.
009    */
010    public class AnimationTester
011    {
012       public static void main(String[] args)
013       {
014          JFrame frame = new JFrame();
015          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
016    
017          ArrayComponent panel = new ArrayComponent();
018          frame.add(panel, BorderLayout.CENTER);
019    
020          JButton stepButton = new JButton("Step");
021          final JButton runButton = new JButton("Run");
022    
023          JPanel buttons = new JPanel();
024          buttons.add(stepButton);
025          buttons.add(runButton);
026          frame.add(buttons, BorderLayout.NORTH);
027          frame.setSize(FRAME_WIDTH, FRAME_HEIGHT);
028          frame.setVisible(true);
029    
030          Double[] values = new Double[VALUES_LENGTH];
031          for (int i = 0; i < values.length; i++)
032             values[i] = Math.random() * panel.getHeight();
033    
034          final BlockingQueue<String> queue = new LinkedBlockingQueue<String>();
035          queue.add("Step");
036    
037          final Sorter sorter = new Sorter(values, panel, queue);
038    
039          stepButton.addActionListener(new
040             ActionListener()
041             {
042                public void actionPerformed(ActionEvent event)
043                {
044                   queue.add("Step");
045                   runButton.setEnabled(true);
046                }
047             });
048    
049          runButton.addActionListener(new
050             ActionListener()
051             {
052                public void actionPerformed(ActionEvent event)
053                {
054                   runButton.setEnabled(false);
055                   queue.add("Run");
056                }
057             });
058    
059          Thread sorterThread = new Thread(sorter);
060          sorterThread.start();
061       }
062    
063       private static final int FRAME_WIDTH = 300;
064       private static final int FRAME_HEIGHT = 300;
065       private static final int VALUES_LENGTH = 30;
066    }
