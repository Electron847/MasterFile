001    package horstmann.ch09_animation2;
002    import java.awt.*;
003    import java.awt.geom.*;
004    import javax.swing.*;
005    
006    /**
007       This component draws an array and marks two elements in the
008       array.
009    */
010    public class ArrayComponent extends JComponent
011    {
012       public synchronized void paintComponent(Graphics g)
013       {
014          if (values == null) return;
015          Graphics2D g2 = (Graphics2D) g;
016          int width = getWidth() / values.length;
017          for (int i = 0; i < values.length; i++)
018          {
019             Double v = values[i];
020             Rectangle2D bar = new Rectangle2D.Double(
021                   width * i, 0, width, v);
022             if (v == marked1 || v == marked2)
023                g2.fill(bar);
024             else
025                g2.draw(bar);
026          }
027       }
028    
029       /**
030          Sets the values to be painted.
031          @param values the array of values to display
032          @param marked1 the first marked element
033          @param marked2 the second marked element
034       */
035       public synchronized void setValues(Double[] values,
036          Double marked1, Double marked2)
037       {
038          this.values = (Double[]) values.clone();
039          this.marked1 = marked1;
040          this.marked2 = marked2;
041          repaint();
042       }
043    
044       private Double marked1;
045       private Double marked2;
046       private Double[] values;
047    }
