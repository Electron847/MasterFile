001    package horstmann.ch10_command;
002    import java.awt.*;
003    import javax.swing.*;
004    
005    /**
006       This program demonstrates action objects. Two actions
007       insert greetings into a text area. Each action can be
008       triggered by a menu item or toolbar button. When an
009       action is carried out, the opposite action becomes enabled.
010    */
011    public class CommandTester
012    {
013       public static void main(String[] args)
014       {
015          JFrame frame = new JFrame();
016          JMenuBar bar = new JMenuBar();
017          frame.setJMenuBar(bar);
018          JMenu menu = new JMenu("Say");
019          bar.add(menu);
020          JToolBar toolBar = new JToolBar();
021          frame.add(toolBar, BorderLayout.NORTH);
022          JTextArea textArea = new JTextArea(10, 40);
023          frame.add(textArea, BorderLayout.CENTER);
024    
025          GreetingAction helloAction = new GreetingAction(
026                "Hello, World", textArea);
027          helloAction.putValue(Action.NAME, "Hello");
028          helloAction.putValue(Action.SMALL_ICON,
029             new ImageIcon("hello.png"));
030    
031          GreetingAction goodbyeAction = new GreetingAction(
032                "Goodbye, World", textArea);
033          goodbyeAction.putValue(Action.NAME, "Goodbye");
034          goodbyeAction.putValue(Action.SMALL_ICON,
035                new ImageIcon("goodbye.png"));
036    
037          helloAction.setOpposite(goodbyeAction);
038          goodbyeAction.setOpposite(helloAction);
039          goodbyeAction.setEnabled(false);
040    
041          menu.add(helloAction);
042          menu.add(goodbyeAction);
043    
044          toolBar.add(helloAction);
045          toolBar.add(goodbyeAction);
046    
047          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
048          frame.pack();
049          frame.setVisible(true);
050       }
051    }
