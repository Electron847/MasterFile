001    package horstmann.ch10_proxy;
002    import java.awt.*;
003    import javax.swing.*;
004    
005    /**
006       A proxy for delayed loading of image icons.
007    */
008    public class ImageProxy implements Icon
009    {
010       /**
011          Constructs a proxy for delayed loading of an image file.
012          @param name the file name
013       */
014       public ImageProxy(String name)
015       {
016          this.name = name;
017          image = null;
018       }
019    
020       public void paintIcon(Component c, Graphics g, int x, int y)
021       {
022          ensureImageLoaded();
023          image.paintIcon(c, g, x, y);
024       }
025    
026       public int getIconWidth()
027       {
028          ensureImageLoaded();
029          return image.getIconWidth();
030       }
031    
032       public int getIconHeight()
033       {
034          ensureImageLoaded();
035          return image.getIconHeight();
036       }
037    
038       /**
039          Loads the image if it hasn't been loaded yet. Prints
040          a message when the image is loaded.
041       */
042       private void ensureImageLoaded()
043       {
044          if (image == null)
045          {
046             System.out.println("Loading " + name);
047             image = new ImageIcon(name);
048          }
049       }
050    
051       private String name;
052       private ImageIcon image;
053    }
