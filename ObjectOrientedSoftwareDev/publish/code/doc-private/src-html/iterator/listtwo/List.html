001    package iterator.listtwo;
002    import java.util.NoSuchElementException;
003      
004    /* public */
005    interface List {
006      public void accept(IntIterator i);
007    }
008    
009    /* public */
010    class ListF {
011      private ListF() {}
012      public static final List nil = new Nil(); /* Singleton */
013      public static final List cons(int hd, List tl) /* Factory */ {
014        return new Cons(hd, tl);
015      }
016    }
017    
018    /* public */
019    interface IntIterator {
020      public void run(int element);
021    }
022    
023    
024    /*
025     *************************************************************************
026     * List classes.
027     *************************************************************************
028     */
029    class Nil implements List {
030      Nil() {}
031      public String toString() { return "nil"; }
032      public void accept(IntIterator i) { }
033    }
034    
035    class Cons implements List {
036      private final int _hd;
037      private final List _tl;
038      Cons(int hd, List tl) { _hd = hd; _tl = tl; }
039      public String toString() { return _hd + "::" + _tl.toString(); }
040      public void accept(IntIterator i) {
041        i.run(_hd);
042        _tl.accept(i);
043      }
044    }
045    
046    /*
047     *************************************************************************
048     * Internal Iterators.
049     * Traversal controlled by the list.
050     *************************************************************************
051     */
052    class Sum implements IntIterator {
053      public int value = 0;
054      public void run(int hd) {
055        value += hd;
056      }
057    }
058    
059    class Reverse implements IntIterator {
060      public List value = ListF.nil;
061      public void run(int hd) {
062        value = ListF.cons(hd, value);
063      }
064    }
065    
066    /*
067     *************************************************************************
068     * A test case.
069     *************************************************************************
070     */
071    public class Main {
072      public static void main(String[] args) {
073        List test = ListF.cons(1, ListF.cons(2, ListF.cons(3, ListF.nil)));
074        System.out.println(test);
075    
076        Sum v1 = new Sum();
077        test.accept(v1);
078        System.out.println(v1.value);
079    
080        Reverse v3 = new Reverse();
081        test.accept(v3);
082        System.out.println(v3.value);
083      }
084    }
