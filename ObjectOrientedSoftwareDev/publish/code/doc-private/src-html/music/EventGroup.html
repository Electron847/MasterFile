001    package music;
002    import java.util.List;
003    import java.util.LinkedList;
004    class EventGroup implements Event {
005      List<Event> _events = new LinkedList<Event>();
006      public void add(Event e) {
007        _events.add(e);
008      }
009      
010      public void play() {
011        for (Event e : _events) {
012          e.play();
013        }
014      }
015    }
