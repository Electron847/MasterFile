004    public class Main2 {
005      public static void main(String args[]) {
006        try {
007          Person person = new Person("Matt", 30);
008          PersonEntry entry = new PersonEntry(person, 1);
009          ObjectOutputStream os
010            = new ObjectOutputStream (new FileOutputStream("out.dat"));
011          os.writeObject(entry);
012          ObjectInputStream is
013            = new ObjectInputStream (new FileInputStream("out.dat"));
014          Object o = is.readObject();
015          PersonEntry entry2 = (PersonEntry) o;
016          System.out.println("Entry restored from file is" + entry2.toString());
017        } catch (Exception e) { e.printStackTrace(); }
018      }

021    class PersonEntry implements Serializable {
022      private static final long serialVersionUID = 2008L;
023      private Person person = null;
024      private int personNumber = 0;
025      
026      public PersonEntry(Person person, int personNumber) {
027        this.person = person;
028        this.personNumber = personNumber;
029      }
030      public Person getPerson() {
031        return person;
032      }
033      public int getPersonNumber() {
034        return personNumber;
035      }
036      public String toString() {
037        return person.toString() +  " Number " + Integer.toString(personNumber);
038      }
039    }
040    
041    class Person implements Serializable {
042      private String name = "";
043      private int age = 0;
044      
045      public Person(String name, int age) {
046        this.name = name;
047        this.age = age;
048      }
049      public String getName() {
050        return name;
051      }
052      public int getAge() {
053        return age;
054      }
055      public String toString() {
056        return "Name: " + name + " Age: " + Integer.toString(age);
057      }
058    }
