001    package serialization.dynamicType;
002    import java.io.*;
003    
004    public class Main {
005      public static void main(String args[])
006        throws FileNotFoundException, IOException, ClassNotFoundException
007      {
008        ObjectOutputStream os
009          = new ObjectOutputStream (new FileOutputStream("out.dat"));
010        Entry eo = new Entry(1, new X0());
011        os.writeObject(eo);
012        
013        ObjectInputStream is
014          = new ObjectInputStream (new FileInputStream("out.dat"));
015        Entry ei = (Entry) is.readObject();
016        System.out.println(ei);
017      }
018    }
019    
020    class Entry implements Serializable {
021      private static final long serialVersionUID = 2008L;
022      private int _i;
023      private X _x;
024      
025      public Entry(int i, X x) { _i = i; _x = x; }
026      public String toString() {
027        return "Entry(" + Integer.toString(_i) + "," + _x + ")";
028      }
029    }
030    
031    interface X {
032      public String toString();
033    }
034    
035    class X0 implements X, Serializable {
036      public String toString() { return "X0"; }
037    }
038    
039    class X1 implements X {
040      public String toString() { return "X1"; }
041    }
