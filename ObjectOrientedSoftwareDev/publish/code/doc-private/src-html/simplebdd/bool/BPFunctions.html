001    package simplebdd.bool;
002    
003    public interface BoolPredFunctions {
004      public String toGraphString (BoolPred p);
005    }
006    
007    class BPFunctions implements BoolPredFunctions {
008      /** @throws ClassCastException if p is not created by BoolPred.factory() */
009      public String toGraphString (BoolPred p) {
010        ((pBoolPred)p).initToGraphString();
011        StringBuilder b = new StringBuilder();
012        ((pBoolPred)p).toGraphString(b);
013        return b.toString();
014      }
015    }
