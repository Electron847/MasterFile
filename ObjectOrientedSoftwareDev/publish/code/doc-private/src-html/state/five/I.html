003    interface I {
004      public int f();
005      public int g();
006      public void changeDirection();
007    }
008    
009    class C implements I {
010      private State _state = new StateMinus();
011      private int _i;
012      private int _j;
013      public int f() {
014        return _state.f();
015      }
016      public int g() {
017        return _state.g();
018      }
019      public void changeDirection() {
020        _state = _state.next();
021      }
022    
023      interface State {
024        public int f();
025        public int g();
026        public State next();
027      }
028      class StateMinus implements State {
029        public int f() {
030          _i -= 32;
031          return _i;
032      }
033        public int g() {
034          _j -= 27;
035          return _j;
036        }
037        public State next() {
038          return new StatePlus();
039        }
040      }
041      class StatePlus implements State {
042        public int f() {
043          _i += 26;
044          return _i;
045        }
046        public int g() {
047          _j += 42;

049        }

051          return new StateMinus();
