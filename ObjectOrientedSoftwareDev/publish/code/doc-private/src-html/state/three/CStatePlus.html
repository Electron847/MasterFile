001    package state.three;
002    
003    interface CState {
004      public int f(C x);
005      public int g(C x);
006    
007      public static final CState MINUS = new CStateMinus();
008      public static final CState PLUS = new CStatePlus();
009    }
010    
011    class CStateMinus implements CState {
012      public int f(C x) {
013        x._i -= 32;
014        return x._i;
015      }
016      public int g(C x) {
017        x._j -= 27;
018        return x._j;
019      }
020    }
021    class CStatePlus implements CState {
022      public int f(C x) {
023        x._i += 26;
024        return x._i;
025      }
026      public int g(C x) {
027        x._j += 42;
028        return x._j;
029      }
030    }
