001    package template.series1;
002    
003    interface Series {
004      void next();
005      void print();
006    }
007    class SeriesFactory {
008      private SeriesFactory() {}
009      public static Series newArith() { return new ArithSeries(); }
010      public static Series newGeom()  { return new GeomSeries(); }
011    }
012    class ArithSeries implements Series {
013      int _x;
014      int _y = 1;
015      public void next()  { _x++; _y = _y+2; }
016      public void print() { System.out.println("x=" + _x + "; y=" + _y); }
017    }
018    class GeomSeries implements Series {
019      int _x;
020      int _y = 1;
021      public void next()  { _x++; _y = _y*2; }
022      public void print() { System.out.println("x=" + _x + "; y=" + _y); }
023    }
024    
025    public class Main {
026      public static void main(String[] args) {
027        Series x = SeriesFactory.newGeom();
028        x.next();
029        x.next();
030        x.print();
031      }
032    }
