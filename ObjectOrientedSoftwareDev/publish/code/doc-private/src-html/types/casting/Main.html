001    package types.casting;
002    // class Number {} 
003    // class Integer extends Number {} 
004    // class Float extends Number {} 
005    
006    public class Main {
007      private Main() {}
008      public static void main(String[] args) {
009        // Upcasts always compile and run ok
010        Number n1 = new Integer(42);          // implicit upcast
011        Number n2 = (Number) new Float(3.14); // explicit upcast
012    
013        // Explicit downcasts always compile, but may cause runtime errors
014    //  Integer i1 = n1;           // implicit downcast: compiler error
015        Integer i2 = (Integer) n1; // explicit downcast: runtime success
016        Integer i3 = (Integer) n2; // explicit downcast: runtime error
017    
018        // Crosscasts are always disallowed by the compiler
019        // So Integer cannot cast to Float
020    //  Float f1 = (Float) i2;          // crosscast: compiler error
021        Float f2 = (Float) (Number) i2; // upcast+downcast: runtime error
022    
023        // Casting changes the declared type, but not the actual type
024        System.out.println("Object n2=" + ((Object) n2));
025        System.out.println("Number n2=" + n2);
026        System.out.println("Float  n2=" + ((Float) n2));
027      }
028    }
