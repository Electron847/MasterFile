001    package types.point3;
002    import java.awt.Color;
003    interface Point {
004      double getX();
005      double getY();
006    }
007    final class CartesianPoint implements Point {
008      
009      private final double _x;
010      private final double _y;
011      private final Color _color; 
012      public CartesianPoint(double x, double y, Color color) {
013        _x = x;
014        _y = y;
015        _color = color;
016      }
017      public double getX() { return _x; }
018      public double getY() { return _y; }
019      public Color getColor() {
020        System.out.println("It's Cartesian!");
021        return _color;
022      }
023      
024    }
025    final class PolarPoint implements Point {
026      
027      private final double _theta;
028      private final double _r;
029      private final Color _color; 
030      public PolarPoint(double theta, double r, Color color) {
031        _theta = theta;
032        _r = r;
033        _color = color;
034      }
035      public double getX() { return _r*Math.cos(_theta); }
036      public double getY() { return _r*Math.sin(_theta); }
037      public Color getColor() {
038        System.out.println("It's Polar!");
039        return _color;
040      }
041      
042    }
043    public class Main {
044      private Main() {}
045      public static void main(String[] args) {
046        CartesianPoint p1 = new CartesianPoint(0,0,Color.RED);
047        CartesianPoint q1 = new CartesianPoint(1,1,Color.BLUE);
048        PolarPoint r1 = new PolarPoint(0,0,Color.RED);
049    
050        Point p2 = p1;
051        Point q2 = q1;
052        Point r2 = r1;
053      }
054    }
