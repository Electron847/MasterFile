001    package util;
002    
003    public class Logger {
004      private Logger() { }
005      private static boolean _on = true;
006      public static void on () { _on = true; }
007      public static void off () { _on = false; }
008      public static void println (String s) {
009        if (_on) {
010          System.out.println (s);
011        }
012      }
013    }
