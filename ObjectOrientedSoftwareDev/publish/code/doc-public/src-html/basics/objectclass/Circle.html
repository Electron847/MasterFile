001    package basics.objectclass;
002    // A static class
003    public class Main {
004      private Main() {}
005      static public void main (String[] args) {
006        Circle c = new Circle();
007        String s = ((c==null) ? "null" : c.toString());
008        System.out.println(s);
009      }
010    }
011    // An object class
012    final class Circle extends Object {
013      public Circle() { super(); }
014      public String toString() { return "Circle"; }
015    }
