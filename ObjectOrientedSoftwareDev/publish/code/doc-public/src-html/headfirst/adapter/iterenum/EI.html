001    package headfirst.adapter.iterenum;
002    
003    import java.util.*;
004    
005    public class EI {
006            public static void main (String args[]) {
007                    Vector<String> v = new Vector<String>(Arrays.asList(args));
008                    Enumeration enumeration = v.elements();
009                    while (enumeration.hasMoreElements()) {
010                            System.out.println(enumeration.nextElement());
011                    }
012                    Iterator iterator = v.iterator();
013                    while (iterator.hasNext()) {
014                            System.out.println(iterator.next());
015                    }
016            }
017    }
