001    package headfirst.combined.djview;
002        
003    import java.awt.*;
004    import java.awt.event.*;
005    import javax.swing.*;
006    
007    public class DJView implements ActionListener,  BeatObserver, BPMObserver {
008            BeatModelInterface model;
009            ControllerInterface controller;
010        JFrame viewFrame;
011        JPanel viewPanel;
012            BeatBar beatBar;
013            JLabel bpmOutputLabel;
014        JFrame controlFrame;
015        JPanel controlPanel;
016        JLabel bpmLabel;
017        JTextField bpmTextField;
018        JButton setBPMButton;
019        JButton increaseBPMButton;
020        JButton decreaseBPMButton;
021        JMenuBar menuBar;
022        JMenu menu;
023        JMenuItem startMenuItem;
024        JMenuItem stopMenuItem;
025    
026        public DJView(ControllerInterface controller, BeatModelInterface model) {   
027                    this.controller = controller;
028                    this.model = model;
029                    model.registerObserver((BeatObserver)this);
030                    model.registerObserver((BPMObserver)this);
031        }
032        
033        public void createView() {
034                    // Create all Swing components here
035            viewPanel = new JPanel(new GridLayout(1, 2));
036            viewFrame = new JFrame("View");
037            viewFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
038            viewFrame.setSize(new Dimension(100, 80));
039            bpmOutputLabel = new JLabel("offline", SwingConstants.CENTER);
040                    beatBar = new BeatBar();
041                    beatBar.setValue(0);
042            JPanel bpmPanel = new JPanel(new GridLayout(2, 1));
043                    bpmPanel.add(beatBar);
044            bpmPanel.add(bpmOutputLabel);
045            viewPanel.add(bpmPanel);
046            viewFrame.getContentPane().add(viewPanel, BorderLayout.CENTER);
047            viewFrame.pack();
048            viewFrame.setVisible(true);
049            }
050      
051      
052        public void createControls() {
053                    // Create all Swing components here
054            JFrame.setDefaultLookAndFeelDecorated(true);
055            controlFrame = new JFrame("Control");
056            controlFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
057            controlFrame.setSize(new Dimension(100, 80));
058    
059            controlPanel = new JPanel(new GridLayout(1, 2));
060    
061            menuBar = new JMenuBar();
062            menu = new JMenu("DJ Control");
063            startMenuItem = new JMenuItem("Start");
064            menu.add(startMenuItem);
065            startMenuItem.addActionListener(new ActionListener() {
066                public void actionPerformed(ActionEvent event) {
067                    controller.start();
068                }
069            });
070            stopMenuItem = new JMenuItem("Stop");
071            menu.add(stopMenuItem); 
072            stopMenuItem.addActionListener(new ActionListener() {
073                public void actionPerformed(ActionEvent event) {
074                    controller.stop();
075                }
076            });
077            JMenuItem exit = new JMenuItem("Quit");
078            exit.addActionListener(new ActionListener() {
079                public void actionPerformed(ActionEvent event) {
080                    System.exit(0);
081                }
082            });
083    
084            menu.add(exit);
085            menuBar.add(menu);
086            controlFrame.setJMenuBar(menuBar);
087    
088            bpmTextField = new JTextField(2);
089            bpmLabel = new JLabel("Enter BPM:", SwingConstants.RIGHT);
090            setBPMButton = new JButton("Set");
091            setBPMButton.setSize(new Dimension(10,40));
092            increaseBPMButton = new JButton(">>");
093            decreaseBPMButton = new JButton("<<");
094            setBPMButton.addActionListener(this);
095            increaseBPMButton.addActionListener(this);
096            decreaseBPMButton.addActionListener(this);
097    
098            JPanel buttonPanel = new JPanel(new GridLayout(1, 2));
099    
100                    buttonPanel.add(decreaseBPMButton);
101                    buttonPanel.add(increaseBPMButton);
102    
103            JPanel enterPanel = new JPanel(new GridLayout(1, 2));
104            enterPanel.add(bpmLabel);
105            enterPanel.add(bpmTextField);
106            JPanel insideControlPanel = new JPanel(new GridLayout(3, 1));
107            insideControlPanel.add(enterPanel);
108            insideControlPanel.add(setBPMButton);
109            insideControlPanel.add(buttonPanel);
110            controlPanel.add(insideControlPanel);
111            
112            bpmLabel.setBorder(BorderFactory.createEmptyBorder(5,5,5,5));
113            bpmOutputLabel.setBorder(BorderFactory.createEmptyBorder(5,5,5,5));
114    
115            controlFrame.getRootPane().setDefaultButton(setBPMButton);
116            controlFrame.getContentPane().add(controlPanel, BorderLayout.CENTER);
117    
118            controlFrame.pack();
119            controlFrame.setVisible(true);
120        }
121    
122            public void enableStopMenuItem() {
123            stopMenuItem.setEnabled(true);
124            }
125    
126            public void disableStopMenuItem() {
127            stopMenuItem.setEnabled(false);
128            }
129    
130            public void enableStartMenuItem() {
131            startMenuItem.setEnabled(true);
132            }
133    
134            public void disableStartMenuItem() {
135            startMenuItem.setEnabled(false);
136            }
137    
138        public void actionPerformed(ActionEvent event) {
139                    if (event.getSource() == setBPMButton) {
140                            int bpm = Integer.parseInt(bpmTextField.getText());
141                    controller.setBPM(bpm);
142                    } else if (event.getSource() == increaseBPMButton) {
143                            controller.increaseBPM();
144                    } else if (event.getSource() == decreaseBPMButton) {
145                            controller.decreaseBPM();
146                    }
147        }
148    
149            public void updateBPM() {
150                    if (model != null) {
151                            int bpm = model.getBPM();
152                            if (bpm == 0) {
153                                    if (bpmOutputLabel != null) {
154                                    bpmOutputLabel.setText("offline");
155                                    }
156                            } else {
157                                    if (bpmOutputLabel != null) {
158                                    bpmOutputLabel.setText("Current BPM: " + model.getBPM());
159                                    }
160                            }
161                    }
162            }
163      
164            public void updateBeat() {
165                    if (beatBar != null) {
166                             beatBar.setValue(100);
167                    }
168            }
169    }
