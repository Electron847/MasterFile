001    package headfirst.command.party;
002    
003    public class StereoOnCommand implements Command {
004            Stereo stereo;
005    
006            public StereoOnCommand(Stereo stereo) {
007                    this.stereo = stereo;
008            }
009    
010            public void execute() {
011                    stereo.on();
012            }
013    
014            public void undo() {
015                    stereo.off();
016            }
017    }
