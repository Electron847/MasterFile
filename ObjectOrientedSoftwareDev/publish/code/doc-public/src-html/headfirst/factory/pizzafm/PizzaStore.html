001    package headfirst.factory.pizzafm;
002    
003    public abstract class PizzaStore {
004     
005            abstract Pizza createPizza(String item);
006     
007            public Pizza orderPizza(String type) {
008                    Pizza pizza = createPizza(type);
009                    System.out.println("--- Making a " + pizza.getName() + " ---");
010                    pizza.prepare();
011                    pizza.bake();
012                    pizza.cut();
013                    pizza.box();
014                    return pizza;
015            }
016    }
