001    package headfirst.iterator.dinermergercafe;
002    
003    import java.util.*;
004    
005    public class CafeMenu implements Menu {
006            Hashtable<String,MenuItem> menuItems = new Hashtable<String,MenuItem>();
007      
008            public CafeMenu() {
009                    addItem("Veggie Burger and Air Fries",
010                            "Veggie burger on a whole wheat bun, lettuce, tomato, and fries",
011                            true, 3.99);
012                    addItem("Soup of the day",
013                            "A cup of the soup of the day, with a side salad",
014                            false, 3.69);
015                    addItem("Burrito",
016                            "A large burrito, with whole pinto beans, salsa, guacamole",
017                            true, 4.29);
018            }
019     
020            public void addItem(String name, String description, 
021                                 boolean vegetarian, double price) 
022            {
023                    MenuItem menuItem = new MenuItem(name, description, vegetarian, price);
024                    menuItems.put(menuItem.getName(), menuItem);
025            }
026     
027            public Hashtable getItems() {
028                    return menuItems;
029            }
030      
031            public Iterator createIterator() {
032                    return menuItems.values().iterator();
033            }
034    }
