003    import java.util.ArrayList;
004    import java.util.Iterator;
005    
006    public class PancakeHouseMenu implements Menu {
007            ArrayList<MenuItem> menuItems;
008     
009            public PancakeHouseMenu() {
010                    menuItems = new ArrayList<MenuItem>();
011        
012                    addItem("K&B's Pancake Breakfast", 
013                            "Pancakes with scrambled eggs, and toast", 
014                            true,
015                            2.99);
016     
017                    addItem("Regular Pancake Breakfast", 
018                            "Pancakes with fried eggs, sausage", 
019                            false,
020                            2.99);
021     
022                    addItem("Blueberry Pancakes",
023                            "Pancakes made with fresh blueberries, and blueberry syrup",
024                            true,
025                            3.49);
026     
027                    addItem("Waffles",
028                            "Waffles, with your choice of blueberries or strawberries",
029                            true,
030                            3.59);
031            }
032    
033            public void addItem(String name, String description,
034                                boolean vegetarian, double price)
035            {
036                    MenuItem menuItem = new MenuItem(name, description, vegetarian, price);
037                    menuItems.add(menuItem);
038            }
039     
040            public ArrayList getMenuItems() {
041                    return menuItems;
042            }
043      
044            public Iterator createIterator() {
045                    return menuItems.iterator();
046            }
