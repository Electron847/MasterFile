001    package headfirst.iterator.transition;
002    import java.util.*;
003      
004         
005    public class Waitress {
006            ArrayList menus;
007         
008      
009            public Waitress(ArrayList menus) {
010                    this.menus = menus;
011            }
012       
013            public void printMenu() {
014                    Iterator menuIterator = menus.iterator();
015                    while(menuIterator.hasNext()) {
016                            Menu menu = (Menu)menuIterator.next();
017                            printMenu(menu.createIterator());
018                    }
019            }
020       
021            void printMenu(Iterator iterator) {
022                    while (iterator.hasNext()) {
023                            MenuItem menuItem = (MenuItem)iterator.next();
024                            System.out.print(menuItem.getName() + ", ");
025                            System.out.print(menuItem.getPrice() + " -- ");
026                            System.out.println(menuItem.getDescription());
027                    }
028            }
029    }
