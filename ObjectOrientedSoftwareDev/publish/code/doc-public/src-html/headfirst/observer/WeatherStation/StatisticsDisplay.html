001    package headfirst.observer.WeatherStation;
002    
003    import java.util.*;
004    
005    public class StatisticsDisplay implements Observer, DisplayElement {
006            private float maxTemp = 0.0f;
007            private float minTemp = 200;
008            private float tempSum= 0.0f;
009            private int numReadings;
010            private WeatherData weatherData;
011    
012            public StatisticsDisplay(WeatherData weatherData) {
013                    this.weatherData = weatherData;
014                    weatherData.registerObserver(this);
015            }
016    
017            public void update(float temp, float humidity, float pressure) {
018                    tempSum += temp;
019                    numReadings++;
020    
021                    if (temp > maxTemp) {
022                            maxTemp = temp;
023                    }
024     
025                    if (temp < minTemp) {
026                            minTemp = temp;
027                    }
028    
029                    display();
030            }
031    
032            public void display() {
033                    System.out.println("Avg/Max/Min temperature = " + (tempSum / numReadings)
034                            + "/" + maxTemp + "/" + minTemp);
035            }
036    }
