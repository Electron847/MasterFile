001    package headfirst.observer.WeatherStationObservable;
002            
003    import java.util.Observable;
004    import java.util.Observer;
005            
006    public class WeatherData extends Observable {
007            private float temperature;
008            private float humidity;
009            private float pressure;
010            
011            public WeatherData() { }
012            
013            public void measurementsChanged() {
014                    setChanged();
015                    notifyObservers();
016            }
017            
018            public void setMeasurements(float temperature, float humidity, float pressure) {
019                    this.temperature = temperature;
020                    this.humidity = humidity;
021                    this.pressure = pressure;
022                    measurementsChanged();
023            }
024            
025            public float getTemperature() {
026                    return temperature;
027            }
028            
029            public float getHumidity() {
030                    return humidity;
031            }
032            
033            public float getPressure() {
034                    return pressure;
035            }
036    }
