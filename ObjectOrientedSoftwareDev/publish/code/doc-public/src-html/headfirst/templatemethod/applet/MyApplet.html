001    package headfirst.templatemethod.applet;
002    
003    import java.applet.Applet;
004    import java.awt.Graphics;
005    
006    public class MyApplet extends Applet {
007        String message;
008     
009        public void init() {
010            message = "Hello World, I'm alive!";
011            repaint();
012        }
013     
014        public void start() {
015            message = "Now I'm starting up...";
016            repaint();
017        }
018     
019        public void stop() {
020            message = "Oh, now I'm being stopped...";
021            repaint();
022        }
023     
024        public void destroy() {
025            message = "Goodbye, cruel world";
026            repaint();
027        }
028     
029        public void paint(Graphics g) {
030            g.drawString(message, 5, 15);
031        }
032    }
