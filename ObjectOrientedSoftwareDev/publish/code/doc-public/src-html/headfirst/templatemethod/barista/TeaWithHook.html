001    package headfirst.templatemethod.barista;
002    
003    import java.io.*;
004    
005    public class TeaWithHook extends CaffeineBeverageWithHook {
006     
007            public void brew() {
008                    System.out.println("Steeping the tea");
009            }
010     
011            public void addCondiments() {
012                    System.out.println("Adding Lemon");
013            }
014     
015            public boolean customerWantsCondiments() {
016    
017                    String answer = getUserInput();
018    
019                    if (answer.toLowerCase().startsWith("y")) {
020                            return true;
021                    } else {
022                            return false;
023                    }
024            }
025     
026            private String getUserInput() {
027                    // get the user's response
028                    String answer = null;
029    
030                    System.out.print("Would you like milk and sugar with your coffee (y/n)? ");
031    
032                    BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
033                    try {
034                            answer = in.readLine();
035                    } catch (IOException ioe) {
036                            System.err.println("IO error trying to read your answer");
037                    }
038                    if (answer == null) {
039                            return "no";
040                    }
041                    return answer;
042            }
043    }
