001    package horstmann.ch04_icon2;
002    import java.awt.*;
003    import java.awt.geom.*;
004    import javax.swing.*;
005    
006    /**
007       An icon that has the shape of the planet Mars.
008    */
009    public class MarsIcon implements Icon
010    {
011       /**
012          Constructs a Mars icon of a given size.
013          @param aSize the size of the icon
014       */
015       public MarsIcon(int aSize)
016       {
017          size = aSize;
018       }
019    
020       public int getIconWidth()
021       {
022          return size;
023       }
024    
025       public int getIconHeight()
026       {
027          return size;
028       }
029    
030       public void paintIcon(Component c, Graphics g, int x, int y)
031       {
032          Graphics2D g2 = (Graphics2D) g;
033          Ellipse2D.Double planet = new Ellipse2D.Double(x, y,
034                size, size);
035          g2.setColor(Color.RED);
036          g2.fill(planet);
037       }
038    
039       private int size;
040    }
