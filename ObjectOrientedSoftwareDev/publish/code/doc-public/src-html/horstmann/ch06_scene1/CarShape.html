001    package horstmann.ch06_scene1;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       A car shape.
007    */
008    public class CarShape extends SelectableShape
009    {
010       /**
011          Constructs a car shape.
012          @param x the left of the bounding rectangle
013          @param y the top of the bounding rectangle
014          @param width the width of the bounding rectangle
015       */
016       public CarShape(int x, int y, int width)
017       {
018          this.x = x;
019          this.y = y;
020          this.width = width;
021       }
022    
023       public void draw(Graphics2D g2)
024       {
025          Rectangle2D.Double body
026             = new Rectangle2D.Double(x, y + width / 6, 
027                width - 1, width / 6);
028          Ellipse2D.Double frontTire
029             = new Ellipse2D.Double(x + width / 6, y + width / 3, 
030                width / 6, width / 6);
031          Ellipse2D.Double rearTire
032             = new Ellipse2D.Double(x + width * 2 / 3, 
033                y + width / 3,
034                width / 6, width / 6);
035    
036          // The bottom of the front windshield
037          Point2D.Double r1
038             = new Point2D.Double(x + width / 6, y + width / 6);
039          // The front of the roof
040          Point2D.Double r2
041             = new Point2D.Double(x + width / 3, y);
042          // The rear of the roof
043          Point2D.Double r3
044             = new Point2D.Double(x + width * 2 / 3, y);
045          // The bottom of the rear windshield
046          Point2D.Double r4
047             = new Point2D.Double(x + width * 5 / 6, y + width / 6);
048          Line2D.Double frontWindshield
049             = new Line2D.Double(r1, r2);
050          Line2D.Double roofTop
051             = new Line2D.Double(r2, r3);
052          Line2D.Double rearWindshield
053             = new Line2D.Double(r3, r4);
054    
055          g2.draw(body);
056          g2.draw(frontTire);
057          g2.draw(rearTire);
058          g2.draw(frontWindshield);
059          g2.draw(roofTop);
060          g2.draw(rearWindshield);
061       }
062       
063       public void drawSelection(Graphics2D g2)
064       {
065          Rectangle2D.Double body
066             = new Rectangle2D.Double(x, y + width / 6, 
067                width - 1, width / 6);
068          g2.fill(body);
069       }
070    
071       public boolean contains(Point2D p)
072       {
073          return x <= p.getX() && p.getX() <= x + width 
074             && y <= p.getY() && p.getY() <= y + width / 2;
075       }
076    
077       public void translate(int dx, int dy)
078       {
079          x += dx;
080          y += dy;
081       }
082    
083       private int x;
084       private int y;
085       private int width;
086    }
