001    package horstmann.ch06_scene3;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       A house shape.
007    */
008    public class HouseShape extends CompoundShape
009    {
010       /**
011          Constructs a house shape.
012          @param x the left of the bounding rectangle
013          @param y the top of the bounding rectangle
014          @param width the width of the bounding rectangle
015       */
016       public HouseShape(int x, int y, int width)
017       {
018          Rectangle2D.Double base 
019             = new Rectangle2D.Double(x, y + width, width, width);
020    
021          // The left bottom of the roof
022          Point2D.Double r1
023             = new Point2D.Double(x, y + width);
024          // The top of the roof
025          Point2D.Double r2
026             = new Point2D.Double(x + width / 2, y);
027          // The right bottom of the roof
028          Point2D.Double r3
029             = new Point2D.Double(x + width, y + width);
030    
031          Line2D.Double roofLeft
032             = new Line2D.Double(r1, r2);
033          Line2D.Double roofRight
034             = new Line2D.Double(r2, r3);
035    
036          add(base);
037          add(roofLeft);
038          add(roofRight);
039       }
040    }
