001    package horstmann.ch08_graphed;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       A circular node that is filled with a color.
007    */
008    public class CircleNode implements Node
009    {
010       /**
011          Construct a circle node with a given size and color.
012          @param aColor the fill color
013       */
014       public CircleNode(Color aColor)
015       {
016          size = DEFAULT_SIZE;
017          x = 0;
018          y = 0;
019          color = aColor;
020       }
021    
022       public Object clone()
023       {
024          try
025          {
026             return super.clone();
027          }
028          catch (CloneNotSupportedException exception)
029          {
030             return null;
031          }
032       }
033    
034       public void draw(Graphics2D g2)
035       {
036          Ellipse2D circle = new Ellipse2D.Double(
037                x, y, size, size);
038          Color oldColor = g2.getColor();
039          g2.setColor(color);
040          g2.fill(circle);
041          g2.setColor(oldColor);
042          g2.draw(circle);
043       }
044    
045       public void translate(double dx, double dy)
046       {
047          x += dx;
048          y += dy;
049       }
050    
051       public boolean contains(Point2D p)
052       {
053          Ellipse2D circle = new Ellipse2D.Double(
054                x, y, size, size);
055          return circle.contains(p);
056       }
057    
058       public Rectangle2D getBounds()
059       {
060          return new Rectangle2D.Double(
061                x, y, size, size);
062       }
063    
064       public Point2D getConnectionPoint(Point2D other)
065       {
066          double centerX = x + size / 2;
067          double centerY = y + size / 2;
068          double dx = other.getX() - centerX;
069          double dy = other.getY() - centerY;
070          double distance = Math.sqrt(dx * dx + dy * dy);
071          if (distance == 0) return other;
072          else return new Point2D.Double(
073                centerX + dx * (size / 2) / distance,
074                centerY + dy * (size / 2) / distance);
075       }
076    
077       private double x;
078       private double y;
079       private double size;
080       private Color color;  
081       private static final int DEFAULT_SIZE = 20;
082    }
