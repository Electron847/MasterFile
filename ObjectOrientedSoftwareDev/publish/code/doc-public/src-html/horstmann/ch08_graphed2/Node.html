001    package horstmann.ch08_graphed2;
002    import java.awt.*;
003    import java.awt.geom.*;
004    import java.io.*;
005    
006    /**
007       A node in a graph.
008    */
009    public interface Node extends Serializable, Cloneable
010    {
011       /**
012          Draw the node.
013          @param g2 the graphics context
014       */
015       void draw(Graphics2D g2);
016    
017       /**
018          Translates the node by a given amount.
019          @param dx the amount to translate in the x-direction
020          @param dy the amount to translate in the y-direction
021       */
022       void translate(double dx, double dy);
023    
024       /**
025          Tests whether the node contains a point.
026          @param aPoint the point to test
027          @return true if this node contains aPoint
028       */
029       boolean contains(Point2D aPoint);
030    
031       /**
032          Get the best connection point to connect this node
033          with another node. This should be a point on the boundary
034          of the shape of this node.
035          @param aPoint an exterior point that is to be joined
036          with this node
037          @return the recommended connection point
038       */
039       Point2D getConnectionPoint(Point2D aPoint);
040    
041       /**
042          Get the bounding rectangle of the shape of this node
043          @return the bounding rectangle
044       */
045       Rectangle2D getBounds();
046    
047       Object clone();
048    }
