001    package horstmann.ch08_umleditor;
002    import java.util.ArrayList;
003    import java.awt.geom.Point2D;
004    
005    /**
006       An edge that is shaped like a line with up to
007       three segments with an arrowhead
008    */
009    public class ClassRelationshipEdge extends SegmentedLineEdge
010    {
011       /**
012          Constructs a straight edge.
013       */
014       public ClassRelationshipEdge()
015       {
016          bentStyle = BentStyle.STRAIGHT;
017       }
018    
019       /**
020          Sets the bentStyle property
021          @param newValue the bent style
022       */
023       public void setBentStyle(BentStyle newValue) { bentStyle = newValue; }
024       /**
025          Gets the bentStyle property
026          @return the bent style
027       */
028       public BentStyle getBentStyle() { return bentStyle; }
029    
030       public ArrayList<Point2D> getPoints()
031       {
032          return bentStyle.getPath(getStart().getBounds(),
033             getEnd().getBounds());
034       }
035    
036       private BentStyle bentStyle;
037    }
