001    package horstmann.ch08_umleditor;
002    import java.awt.*;
003    import java.awt.geom.*;
004    
005    /**
006       An inivisible node that is used in the toolbar to draw an
007       edge.
008    */
009    public class PointNode implements Node
010    {
011       /**
012          Constructs a point node with coordinates (0, 0)
013       */
014       public PointNode()
015       {
016          point = new Point2D.Double();
017       }
018    
019       public void draw(Graphics2D g2)
020       {
021       }
022    
023       public void translate(double dx, double dy)
024       {
025          point.setLocation(point.getX() + dx,
026             point.getY() + dy);
027       }
028    
029       public boolean contains(Point2D p)
030       {
031          return false;
032       }
033    
034       public Rectangle2D getBounds()
035       {
036          return new Rectangle2D.Double(point.getX(), 
037             point.getY(), 0, 0);
038       }
039    
040       public Point2D getConnectionPoint(Point2D other)
041       {
042          return point;
043       }
044    
045       public Object clone()
046       {
047          try
048          {
049             return super.clone();
050          }
051          catch (CloneNotSupportedException exception)
052          {
053             return null;
054          }
055       }
056    
057       private Point2D point;
058    }
