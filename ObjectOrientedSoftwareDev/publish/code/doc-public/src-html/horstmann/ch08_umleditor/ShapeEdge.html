001    package horstmann.ch08_umleditor;
002    import java.awt.BasicStroke;
003    import java.awt.Graphics2D;
004    import java.awt.Shape;
005    import java.awt.geom.Line2D;
006    import java.awt.geom.Point2D;
007    import java.awt.geom.Rectangle2D;
008    
009    /**
010       A class that assumes that an edge can yield its shape
011       and then takes advantage of the fact that containment testing can 
012       be done by stroking the shape with a fat stroke.
013    */
014    public abstract class ShapeEdge extends AbstractEdge
015    {  
016       /**
017          Returns the path that should be stroked to
018          draw this edge. The path does not include
019          arrow tips or labels.
020          @return a path along the edge
021       */
022       public abstract Shape getShape();
023    
024       public Rectangle2D getBounds(Graphics2D g2)
025       {
026          return getShape().getBounds();
027       }
028    
029       public boolean contains(Point2D aPoint)
030       {
031          final double MAX_DIST = 3;
032    
033          // the end points may contain small nodes, so don't
034          // match them
035          Line2D conn = getConnectionPoints();
036          if (aPoint.distance(conn.getP1()) <= MAX_DIST 
037             || aPoint.distance(conn.getP2()) <= MAX_DIST)
038             return false;
039    
040          Shape p = getShape();
041          BasicStroke fatStroke = new BasicStroke(
042             (float)(2 * MAX_DIST));
043          Shape fatPath = fatStroke.createStrokedShape(p);
044          return fatPath.contains(aPoint);
045       }
046    }
