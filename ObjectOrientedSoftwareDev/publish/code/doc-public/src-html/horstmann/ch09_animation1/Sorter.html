001    package horstmann.ch09_animation1;
002    import java.util.*;
003    
004    /**
005       This runnable executes a sort algorithm.
006       When two elements are compared, the algorithm
007       pauses and updates a panel.
008    */
009    public class Sorter implements Runnable
010    {
011       /**
012          Constructs the sorter.
013          @param values the array to sort
014          @param panel the panel for displaying the array
015       */
016       public Sorter(Double[] values, ArrayComponent panel)
017       {
018          this.values = values;
019          this.panel = panel;
020       }
021    
022       public void run()
023       {
024          Comparator<Double> comp = new
025             Comparator<Double>()
026             {
027                public int compare(Double d1, Double d2)
028                {
029                   panel.setValues(values, d1, d2);
030                   try
031                   {
032                      Thread.sleep(DELAY);
033                   }
034                   catch (InterruptedException exception)
035                   {
036                      Thread.currentThread().interrupt();
037                   }
038                   return (d1).compareTo(d2);
039                }
040             };
041          MergeSorter.sort(values, comp);
042          panel.setValues(values, null, null);
043       }
044    
045       private Double[] values;
046       private ArrayComponent panel;
047       private static final int DELAY = 100;
048    }
