001    package horstmann.ch10_command;
002    import java.awt.event.*;
003    import javax.swing.*;
004    
005    /**
006       This action places a greeting into a text field 
007       and afterwards disables itself and enables its 
008       opposite action.
009    */
010    public class GreetingAction extends AbstractAction
011    {
012       /**
013          Constructs a greeting action.
014          @param greeting the string to add to the text area
015          @param textArea the text area to which to add the greeting
016       */      
017       public GreetingAction(String greeting, JTextArea textArea)
018       {
019          this.greeting = greeting;
020          this.textArea = textArea;
021       }
022    
023       /**
024          Sets the opposite action.
025          @param action the action to be enabled after this action was
026          carried out
027       */
028       public void setOpposite(Action action)
029       {
030          oppositeAction = action;
031       }
032    
033       public void actionPerformed(ActionEvent event)
034       {
035          textArea.append(greeting);
036          textArea.append("\n");
037          if (oppositeAction != null)
038          {
039             setEnabled(false);
040             oppositeAction.setEnabled(true);
041          }
042       }
043    
044       private String greeting;
045       private JTextArea textArea;
046       private Action oppositeAction;
047    }
