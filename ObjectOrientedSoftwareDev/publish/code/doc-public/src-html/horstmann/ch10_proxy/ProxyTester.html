001    package horstmann.ch10_proxy;
002    import java.awt.*;
003    import javax.swing.*;
004    
005    /**
006       This program demonstrates the use of the image proxy.
007       Images are only loaded when you press on a tab.
008    */
009    public class ProxyTester
010    {
011       public static void main(String[] args)
012       {
013          JTabbedPane tabbedPane = new JTabbedPane();
014          for (String name : imageNames)
015          {
016             JLabel label = new JLabel(new ImageProxy(name));
017             tabbedPane.add(name, label);
018          }
019    
020          JFrame frame = new JFrame();
021          frame.add(tabbedPane);
022    
023          frame.setSize(FRAME_WIDTH, FRAME_HEIGHT);
024          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
025          frame.setVisible(true);
026       }
027    
028       private static final String[] imageNames =
029       {
030          "devonian.gif",
031          "permian.gif",
032          "jurassic1.gif",
033          "jurassic2.gif",
034          "cretaceous1.gif",
035          "cretaceous2.gif",
036          "cretaceous3.gif",
037          "eocene1.gif",
038          "eocene2.gif",
039          "oligocene.gif",
040          "miocene.gif",
041          "pleistocene.gif"
042       };
043    
044       private static final int FRAME_WIDTH = 500;
045       private static final int FRAME_HEIGHT = 300;
046    }
