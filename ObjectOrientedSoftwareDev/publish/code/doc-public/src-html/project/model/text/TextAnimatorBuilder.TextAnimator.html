001    package project.model.text;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    import java.util.Observable;
006    import java.util.Observer;
007    import project.model.AnimatorBuilder;
008    import project.model.Car;
009    import project.model.Light;
010    import project.model.MP;
011    import project.model.Road;
012    import project.util.Animator;
013    
014    /** 
015     * A class for building Animators.
016     */
017    public class TextAnimatorBuilder implements AnimatorBuilder {
018      TextAnimator _animator;
019      public TextAnimatorBuilder() {
020        _animator = new TextAnimator();
021      }
022      public Animator getAnimator() {
023        if (_animator == null) { throw new IllegalStateException(); }
024        Animator returnValue = _animator;
025        _animator = null;
026        return returnValue;
027      }
028      public void addLight(Light d, int i, int j) {
029        _animator.addLight(d,i,j);
030      }
031      public void addHorizontalRoad(Road l, int i, int j, boolean eastToWest) {
032        _animator.addRoad(l,i,j);
033      }
034      public void addVerticalRoad(Road l, int i, int j, boolean southToNorth) {
035        _animator.addRoad(l,i,j);
036      }
037    
038    
039      /** Class for drawing the Model. */
040      private static class TextAnimator implements Animator {
041    
042        /** Triple of a model element <code>x</code> and coordinates. */
043        private static class Element<T> {
044          T x;
045          int i;
046          int j;
047          Element(T x, int i, int j) {
048            this.x = x;
049            this.i = i;
050            this.j = j;
051          }
052        }
053        
054        private List<Element<Road>> _roadElements;
055        private List<Element<Light>> _lightElements;
056        TextAnimator() {
057          _roadElements = new ArrayList<Element<Road>>();
058          _lightElements = new ArrayList<Element<Light>>();
059        }    
060        void addLight(Light x, int i, int j) {
061          _lightElements.add(new Element<Light>(x,i,j));
062        }
063        void addRoad(Road x, int i, int j) {
064          _roadElements.add(new Element<Road>(x,i,j));
065        }
066        
067        public void dispose() { }
068        
069        public void update(Observable o, Object arg) {
070          for (Element<Light> e : _lightElements) {
071            System.out.print("Light at (" + e.i + "," + e.j + "): ");
072            if (e.x.getState()) {
073              System.out.println("Blue");
074            } else {
075              System.out.println("Yellow");
076            }
077          }
078          for (Element<Road> e : _roadElements) {
079            for (Car d : e.x.getCars()) {
080              System.out.print("Road at (" + e.i + "," + e.j + "): ");
081              System.out.println("Car at " + d.getPosition());
082            }
083          }
084        }
085      }
086    }
