001    package serialization.custom;

003    import java.io.Serializable;
004    import java.io.IOException;

006    import java.io.FileOutputStream;
007    import java.io.ObjectOutputStream;

009    import java.io.ObjectInputStream;
010    
011    class IntList1 implements IntList
012    {
013      private LinkedList<Integer> _v;
014      private void setV(LinkedList<Integer> v) { _v = _v; }
015    
016      public IntList1()               { _v = new LinkedList<Integer>(); }
017      public void addBack(int i)      { _v.addLast(i); }
018      public void addFront(int i)     { _v.addFirst(i); }
019      public int removeFront()        { return _v.removeFirst(); }
020      public int removeBack()         { return _v.removeLast(); }
021      public boolean isEmpty()        { return _v.size() == 0; }
022    }
023    
024    public class Main1 {
025      public static void main(String[] args)
026        throws IOException, FileNotFoundException, ClassNotFoundException 
027      {
028        IntList L = new IntList1();
029        for (int i=0; i<1000; i++)
030          L.addFront(i);
031        
032        ObjectOutputStream os = new ObjectOutputStream (new FileOutputStream("out1.dat"));
033        os.writeObject(L);
