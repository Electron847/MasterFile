001    package serialization.fixedJavaSoftExample;
002    import java.io.*;
003    
004    public class CardReader {
005        public static void main(String[] args) {
006            Card2 card = null;
007    
008            try {
009                FileInputStream in = new FileInputStream("card.out");
010                ObjectInputStream ois = new ObjectInputStream(in);
011                card = (Card2)(ois.readObject());
012            } catch (Exception e) {
013                System.out.println("Problem serializing: " + e);
014            }
015    
016            System.out.println("Card read is: " + card);
017    
018        }
019    }
