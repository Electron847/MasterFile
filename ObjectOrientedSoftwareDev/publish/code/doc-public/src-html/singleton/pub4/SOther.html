001    package singleton.pub4;
002    public abstract class S {
003      private static S _instance;
004      static {
005        if ("linux".equals(System.getProperty("os.name")))
006          _instance = new SLinux();
007        else
008          _instance = new SOther();
009      }

012    }
013    final class SLinux extends S { 
014      private int i;
015      public int inc() {return ++i;}
016    }
017    final class SOther extends S {
018      private int i;
