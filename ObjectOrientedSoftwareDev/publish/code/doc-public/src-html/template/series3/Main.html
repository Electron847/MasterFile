001    package template.series3;
002    
003    interface Series {
004      void next();
005      void print();
006    }
007    class SeriesFactory {
008      private SeriesFactory() {}
009      public static Series newArith() { return new ConcreteSeries(new OpAdd()); }
010      public static Series newGeom()  { return new ConcreteSeries(new OpMul()); }
011    }
012    class ConcreteSeries implements Series {
013      int _x;
014      int _y = 1;
015      public void next()  { _x++; _y = _op.eval(_y,2); }
016      public void print() { System.out.println("x=" + _x + "; y=" + _y); }
017      Op _op;
018      ConcreteSeries(Op op) { _op = op; }
019    }
020    interface Op { public int eval(int x, int y); }
021    class OpAdd implements Op { public int eval(int x, int y) { return x+y; } }
022    class OpMul implements Op { public int eval(int x, int y) { return x*y; } }
023    
024    public class Main {
025      public static void main(String[] args) {
026        Series x = SeriesFactory.newGeom();
027        x.next();
028        x.next();
029        x.print();
030      }
031    }
