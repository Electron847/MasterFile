001    package types.point2;
002    import java.awt.Color;
003    final class PolarPoint {
004      private final double _theta;
005      private final double _r;
006      private final Color _color; 
007      public PolarPoint(double theta, double r, Color color) {
008        _theta = theta;
009        _r = r;
010        _color = color;
011      }
012      public double getX() { return _r*Math.cos(_theta); }
013      public double getY() { return _r*Math.sin(_theta); }
014      public Color getColor() {
015        System.out.println("It's Polar!");
016        return _color;
017      }
018    }
019    public class Main {
020      private Main() {}
021      public static void main(String[] args) {
022        PolarPoint r1 = new PolarPoint(0,0,Color.RED);
023      }
